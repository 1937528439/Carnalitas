# Basic Interaction Effect
carn_spend_time_with_effect = {
    $PLAYER$ = {
        save_scope_as = initiator
    }
    $TARGET$ = {
        save_scope_as = target
    }

    show_as_tooltip = {
        scope:initiator = {
            custom_tooltip = carn_spend_time_with_effect.player
        }
        scope:target = {
            custom_tooltip = carn_spend_time_with_effect.target
        }
    }
}
